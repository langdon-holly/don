Allowed special characters:

whitespace    used : ends name, ignored elsewhere
`             used : character literal
~           unused
@           unused
#             used : end-of-line comment
$           unused
(             used : start delimiter (function call)
)             used : end delimiter (function call)
[             used : start delimiter (list)
]             used : end delimiter (list)
{             used : start delimiter (map)
}             used : end delimiter (map)
\             used : escapes in identifier
|             used : delimits identifier
;             used : comment expression
"             used : quote
,           unused
